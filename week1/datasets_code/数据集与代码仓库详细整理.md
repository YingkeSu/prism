- ✅ **成熟文档**: 详细的使用指南和API

**使用方式**：
```python
from multimodalfilter.multimodal import CrossModalFusion
from torchfilter.filters import UKF

# 创建可微分融合器
fusion = CrossModalFusion(
    modalities=['lidar', 'rgb', 'imu'],
    fusion_type='adaptive'  # 自适应融合
    filter_class=UKF  # 无迹卡尔曼滤波器
)

# 定义UKF滤波器参数
ukf_params = {
    'x': 0.1,  # LiDAR位置
    'P': 1e-5,   # 位置协方差
    'R': 0.05,   # 旋转速率协方差
}

# 创建滤波器实例
ukf = UKF(**ukf_params)

# 融合多模态数据
fused_state, fusion_cov = fusion(
    state_dict={'lidar': lidar_state, 'rgb': rgb_state, 'imu': imu_state},
    observation_noise_dict={'lidar': lidar_cov, 'rgb': rgb_cov, 'imu': imu_cov}
)
```

```python
# 1. 创建自定义特征提取器
# networks/feature_extractors/uav_multimodal_extractor.py

from stable_baselines3.common.torch_layers import BaseFeaturesExtractor
import torch.nn as nn

class UAVMultimodalExtractor(BaseFeaturesExtractor):
    def __init__(self, observation_space, features_dim=512):
        super().__init__(observation_space, features_dim)
        
        # LiDAR编码器
        self.lidar_encoder = PointNetEncoder(
            observation_space["lidar"].shape[1],  # 点云维度
            features_dim=128
        )
        
        # RGB编码器
        self.rgb_encoder = ResNet18Encoder(
            observation_space["rgb"].shape,
            features_dim=256
        )
        
        # IMU编码器
        self.imu_encoder = SimpleMLP(
            observation_space["imu"].shape[1],
            features_dim=128
        )
        
        # 融合层
        self.fusion_net = nn.Sequential(
            nn.Linear(128 + 256 + 128, 512),
            nn.ReLU(),
            nn.Linear(512, 256)
        )
    
    def forward(self, observations):
        # 编码各模态
        lidar_feat = self.lidar_encoder(observations["lidar"])
        rgb_feat = self.rgb_encoder(observations["rgb"])
        imu_feat = self.imu_encoder(observations["imu"])
        
        # 融合特征
        combined = torch.cat([lidar_feat, rgb_feat, imu_feat], dim=1)
        
        # 通过融合层
        return self.fusion_net(combined)

# 2. 创建多模态环境
# envs/uav_multimodal_env.py

import gymnasium as gym
from gymnasium import spaces

class UAVMultimodalEnv(gym.Env):
    def __init__(self):
        super().__init__()
        
        # 定义多模态观测空间
        self.observation_space = spaces.Dict({
            "lidar": spaces.Box(0, 100, 1000, 3),  # 点云
            "rgb": spaces.Box(0, 255, 128, 128, 3),  # RGB图像
            "imu": spaces.Box(-10, 10, 6, float),  # IMU数据
        })
        
        self.action_space = spaces.Box(-1, 1, 4, float)  # 速度控制
        
    def reset(self, seed=None):
        return self._get_observation()
    
    def step(self, action):
        # 更新状态
        observation, reward, done, info = self._uav_dynamics(action)
        return observation, reward, done, info
    
    def _get_observation(self):
        # 模拟多模态传感器数据
        return {
            "lidar": self._lidar_sensor(),
            "rgb": self._rgb_camera(),
            "imu": self._imu_sensor()
        }

# 3. 训练SAC
# train_uav_multimodal.py

from stable_baselines3 import SAC
from networks.feature_extractors.uav_multimodal_extractor import UAVMultimodalExtractor
from envs.uav_multimodal_env import UAVMultimodalEnv

# 创建环境
env = UAVMultimodalEnv()

# 创建模型
model = SAC(
    "MultiInputPolicy",
    env,
    policy_kwargs={
        "features_extractor_class": UAVMultimodalExtractor,
        "features_extractor_kwargs": {"features_dim": 512},
        "net_arch": [256, 256]
    },
    learning_rate=3e-4,
    buffer_size=1000000,
    verbose=1
)

# 训练
model.learn(total_timesteps=1_000_000)
```
